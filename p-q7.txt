class person {
    String name;
    int year;
    person(String name, int year){
        this.name=name;
        this.year = year;
    }
}
class staff extends person{
    double salary;
    staff(String name, int year , int salary){
        this.name= name;
        this.salary= salary;
        this.year = year;
    }
}
class student extends person{
    String department ;
     int attendance;
    student (String name, int year, String department, int attendance){
        this.name = name;
        this.year = year;
        this.department = department;
        this.attendance= attendance;
    }
}

  int salary(){
